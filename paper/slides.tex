% ============================================================
\begin{frame}{The Problem}
\begin{itemize}
    \item AI agents increasingly operate in \textbf{strategic settings}
          --- negotiation, resource allocation, multi-agent coordination
    \item No standardized way to measure:
    \begin{itemize}
        \item Can they \textbf{cooperate}?
        \item Do they \textbf{exploit} weaker opponents?
        \item Are outcomes \textbf{fair}?
        \item Do they \textbf{adapt} to different opponents?
    \end{itemize}
    \item Existing benchmarks conflate language understanding with strategic choice
    \item We need a \textbf{minimal, formal} evaluation harness
\end{itemize}
\end{frame}

% Slide 7: Metrics
% ============================================================
\begin{frame}{Metrics}
All normalized to $[0, 1]$:
\vspace{0.5em}
\begin{description}
    \item[Cooperation Rate ($M_C$)]
        Fraction of cooperative actions across all games
    \item[Exploitation Resistance ($M_E$)]
        Score against always-defect, relative to best/worst
    \item[Pareto Efficiency ($M_P$)]
        How often joint outcomes match the maximum possible
    \item[Fairness Index ($M_F$)]
        Payoff equality between agent and opponent
    \item[Adaptability ($M_A$)]
        Variation in cooperation rate across opponents
    \item[Strategic Reasoning ($M_S$)]
        Composite mean of the five metrics above
\end{description}
\end{frame}

    \item \textbf{Parameters}\\
